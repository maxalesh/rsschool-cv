function strCount(obj){
  let count = 0;
  if (!obj) return count; 
  let values = Object.values(obj);
  for (let item of values) {
    if (typeof(item) === 'object') count += strCount(item);
    else if (typeof(item) === 'string') count += 1
  };
  return count
}
